===
open form page
%require
===
plwr -S plwr-test open "http://localhost:8599/form.html"
---

===
wait for modifier target to load
%require
===
plwr -S plwr-test wait '#mod-target'
---

===
plain click shows none
===
plwr -S plwr-test click '#mod-target'
plwr -S plwr-test text '#mod-result'
---
none

===
alt-click
===
plwr -S plwr-test click '#mod-target' --alt
plwr -S plwr-test text '#mod-result'
---
alt

===
shift-click
===
plwr -S plwr-test click '#mod-target' --shift
plwr -S plwr-test text '#mod-result'
---
shift

===
control-click (macOS only: ctrl+click triggers contextmenu → shows as right)

===

plwr -S plwr-test text '#mod-result'
---
right
