===
wait for select to load
%require
===
plwr -S plwr-test wait '#color'
---

===
select single option by value
===
plwr -S plwr-test select '#color' green
---

===
verify selected value via change handler
===
plwr -S plwr-test text '#color-result'
---
green

===
verify selected value via input-value
===
plwr -S plwr-test input-value '#color'
---
green

===
select different option by value
===
plwr -S plwr-test select '#color' blue
---

===
verify new selection
===
plwr -S plwr-test text '#color-result'
---
blue

===
plwr -S plwr-test select '#color' --label 'Red'
---

===
verify label selection
===
plwr -S plwr-test input-value '#color'

red
